                                                                      # GET 列出元信息

                                                                      GET /api/admin/meta/list

                                                                      # 请求参数

                                                                      名称位置类型必选说明
                                                                      pagequerystring否页数
                                                                      per_pagequerystring否每页个数
                                                                      Authorizationheaderstring是none

                                                                      # 返回示例

                                                                      成功

                                                                      {
  "code": 200,
  "message": "success",
  "data": {
    "content": [
      {
        "id": 1,
        "path": "/a",
        "password": "i",
        "p_sub": false,
        "write": false,
        "w_sub": false,
        "hide": "",
        "h_sub": false,
        "readme": "",
        "r_sub": false
      }
    ],
    "total": 1
  }
}

                                                                      # 请求参数

                                                                      名称位置类型必选中文名说明
                                                                      Authorizationheaderstring是none
                                                                      bodybodyobject否none
                                                                      » idbodyinteger是idnone
                                                                      » pathbodystring是路径none
                                                                      » passwordbodystring是密码none
                                                                      » p_subbodyboolean是密码是否应用到子文件夹none
                                                                      » writebodyboolean是开启写入none
                                                                      » w_subbodyboolean是开启写入是否应用到子文件夹none
                                                                      » hidebodystring是隐藏none
                                                                      » h_subbodyboolean是隐藏是否应用到子文件夹none
                                                                      » readmebodystring是说明none
                                                                      » r_subbodyboolean是说明是否应用到子文件夹none
